import time
import random

الوحوش = ["ذئب شرير", "تنين ناري", "شبح مخيف", "غول قوي"]
الأحوال = ["مطر غزير", "ضباب كثيف", "عاصفة رعدية", "ليل مظلم"]
الغابات = ["غابة مظلمة", "غابة غريبة", "غابة مسكونة", "غابة هادئة"]
النقاط = 0

def طب_ببطء(النص):
    """تعرض النص مع تأخير لمدة ثانية واحدة لتحسين تجربة المستخدم"""
    print(النص)
    time.sleep(1)

def الحصول_على_إدخال_رقمي(الرسالة, الخيارات_الصحيحة):
    """
    تحصل على إدخال رقمي من المستخدم مع التحقق من صحته
    Args:
        الرسالة: رسالة الطلب التي تظهر للمستخدم
        الخيارات_الصحيحة: قائمة بالقيم المقبولة
    Returns:
        الإدخال الصحيح من المستخدم
    """
    while True:
        الإدخال = input(الرسالة)
        if الإدخال in الخيارات_الصحيحة:
            return الإدخال
        طب_ببطء(f"خطأ! الرجاء إدخال أحد الخيارات التالية: {', '.join(الخيارات_الصحيحة)}")

def عرض_الحالة():
    """تعرض حالة اللاعب الحالية ونقاطه"""
    طب_ببطء(f"\n=== نقاطك الحالية: {النقاط} ===")

def إنشاء_سيناريو():
    """تولد سيناريو عشوائي للعبة"""
    return {
        "وحش": random.choice(الوحوش),
        "طقس": random.choice(الأحوال),
        "غابة": random.choice(الغابات)
    }

def التعامل_مع_المواجهة(وحش):
    """تتعامل مع سيناريو مواجهة الوحش"""
    global النقاط
    
    طب_ببطء(f"\nدخلت الغابة.. وفجأة ظهر {وحش}!")
    طب_ببطء("1. تهرب بسرعة")
    طب_ببطء("2. تواجهه بشجاعة")
    
    اختيار = الحصول_على_إدخال_رقمي("ماذا ستفعل؟ (1/2): ", ["1", "2"])
    
    if اختيار == "1":
        طب_ببطء("\nنجحت في الهروب بذكاء!")
        النقاط += 10
    else:
        طب_ببطء("\nواجهته لكنه هزمك!")
        النقاط -= 5

def لعب_جولة():
    """تشغيل جولة كاملة من اللعبة"""
    global النقاط
    
    سيناريو = إنشاء_سيناريو()
    
    طب_ببطء(f"\nأنت أمام {سيناريو['غابة']} والجو {سيناريو['طقس']}.. ماذا تفعل؟")
    طب_ببطء("1. أدخل الغابة")
    طب_ببطء("2. عد للقرية")
    
    اختيار = الحصول_على_إدخال_رقمي("اختر 1 أو 2: ", ["1", "2"])
    
    if اختيار == "1":
        التعامل_مع_المواجهة(سيناريو['وحش'])
    else:
        طب_ببطء("\nعدت للقرية بسلام.")

def بدء_اللعبة():
    """تشغيل اللعبة الرئيسية"""
    طب_ببطء("مرحبًا في لعبة المغامرة!")
    
    while True:
        لعب_جولة()
        عرض_الحالة()
        
        طب_ببطء("\nأتريد اللعب مرة أخرى؟")
        طب_ببطء("1. نعم")
        طب_ببطء("2. لا")
        
        اختيار = الحصول_على_إدخال_رقمي("اختر: ", ["1", "2"])
        if اختيار != "1":
            طب_ببطء(f"\nشكرًا للعب! نقاطك النهائية: {النقاط}")
            break

if __name__ == "__main__":
    بدء_اللعبة()
