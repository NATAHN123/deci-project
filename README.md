import time
import random

الوحوش = ["ذئب شرير", "تنين ناري", "شبح مخيف", "غول قوي"]
الأحوال = ["مطر غزير", "ضباب كثيف", "عاصفة رعدية", "ليل مظلم"]
الغابات = ["غابة مظلمة", "غابة غريبة", "غابة مسكونة", "غابة هادئة"]

النقاط = 0

def طب_ببطء(النص):
    print(النص)
    time.sleep(1)

def لعب_جولة():
    global النقاط
    
    وحش = random.choice(الوحوش)
    طقس = random.choice(الأحوال)
    غابة = random.choice(الغابات)
    
    طب_ببطء(f"\nأنت أمام {غابة} والجو {طقس}.. ماذا تفعل؟")
    طب_ببطء("1. أدخل الغابة")
    طب_ببطء("2. عد للقرية")
    
    while True:
        اختيار = input("اختر 1 أو 2: ")
        
        if اختيار == "1":
            طب_ببطء(f"\nدخلت الغابة.. وفجأة ظهر {وحش}!")
            طب_ببطء("1. تهرب بسرعة")
            طب_ببطء("2. تواجهه بشجاعة")
            
            اختيار2 = input("ماذا ستفعل؟ ")
            
            if اختيار2 == "1":
                طب_ببطء("\nنجحت في الهروب بذكاء!")
                النقاط += 10
            else:
                طب_ببطء("\nواجهته لكنه هزمك!")
                النقاط -= 5
            break
            
        elif اختيار == "2":
            طب_ببطء("\عدت للقرية بسلام.")
            break
            
        else:
            طب_ببطء("اختيار غير صحيح!")

طب_ببطء("مرحبًا في لعبة المغامرة!")

while True:
    لعب_جولة()
    طب_ببطء(f"نقاطك الآن: {النقاط}")
    
    طب_ببطء("\nأتريد اللعب مرة أخرى؟")
    طب_ببطء("1. نعم")
    طب_ببطء("2. لا")
    
    if input("اختر: ") != "1":
        طب_ببطء(f"\nشكرًا للعب! نقاطك النهائية: {النقاط}")
        break
